SELECT p.*, t.*, r.*, pr.*
FROM pedido p 
INNER JOIN turno t ON p.turno = t.idTurno
INNER JOIN tipoRacao r ON p.tipoRacao = r.idRacao
INNER JOIN produtor pr ON p.produtor = pr.idProdutor;
